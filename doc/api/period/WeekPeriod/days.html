@override
List<DayPeriod> get days {
  const generator = DayGenerator();
  final days = <DayPeriod>[];
  var last = generator.of(start);
  while (last.start.isBefore(end)) {
    days.add(last);
    last = generator.after(last);
  }
  return days;
}
